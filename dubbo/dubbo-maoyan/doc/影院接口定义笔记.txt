2、条件列表 
	接口：根据条件获取品牌列表，获取行政区域列表，获取影厅类型列表
	入参：brandId,hallType,areaId
	出参：
					获取品牌 brand
						brandId
						brandName
						isActive
					获取行政区域 area
						areaId
						areaName
						isActive
					获取影厅类型 halltype
						halltypeId
						halltypeName
						isActive
